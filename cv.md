function spinWords(string) {
let newStr = string.split(' ')
    let spin = ''
    for (let i = 0; i < newStr.length; i++) {
        let words = newStr[i]
        if (words.length >= 5) {
            spin = ''
            for (let j = words.length - 1; j >= 0; j--) {
                spin += words[j]
            }
            newStr[i] = spin
        }
    }
    return newStr.join(' ')
}
